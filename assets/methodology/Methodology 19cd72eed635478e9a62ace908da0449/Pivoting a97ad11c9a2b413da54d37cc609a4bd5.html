Pivoting

Pivot Tools:

ssh:

ssh -N -D 80 -o UserKnownHostsFile=/dev/null -o StrictHostKeyChecking=no user@0.0.0.0 -p 22
ssh -N -D 80 -o UserKnownHostsFile=/dev/null -o StrictHostKeyChecking=no user@0.0.0.0 -p 22000 -o KexAlgorithms=diffie-hellman-group14-sha1 -oHostKeyAlgorithms=+ssh-rsa

reverse:

 sudo ssh -o "UserKnownHostsFile=/dev/null" -o "StrictHostKeyChecking=no" -N -R 5555:127.0.0.1:5555 student@192.168.137.52 -p 2222

local port forward:

 sudo ssh -o "UserKnownHostsFile=/dev/null" -o "StrictHostKeyChecking=no" -N -L 9001:127.0.0.1:34023 student@192.168.137.52 -p 2222

Dynamic:

sudo ssh -o "UserKnownHostsFile=/dev/null" -o "StrictHostKeyChecking=no" -N -D 8080 student@192.168.137.52 -p 2222

sshuttle:

sshuttle -r root@0.0.1.1 10.10.10.0/24
