LDAP (389)

  • https://book.hacktricks.xyz/network-services-pentesting/pentesting-ldap
  • cn=admin,dc=example,dc=com
#new syntax
ldapsearch -x -H ldap://0.0.0.0 -b "dc=name1,dc=name2"
#Old syntax
ldapsearch -h <target_system> -p <ldap_port> -b "<base_dn>" "(&(objectclass=*)(userPassword=*))"
ldapsearch -h <target_system> -p <ldap_port> -b "<base_dn>" -D "<dn>" -w "<password>" "(&(objectclass=*)(userPassword=*))"
ldapsearch -h <target_system> -p <ldap_port> -b "<base_dn>"
#Uses anonymous credentials
#nmap ldap enumeration
nmap -n -sV --script "ldap* and not brute" 0.0.0.0 
#ldapdomaindump
ldapdomaindump -h <target_system> -p <ldap_port> -D "<bind_dn>" -w "<bind_password>" -b "<base_dn>"
